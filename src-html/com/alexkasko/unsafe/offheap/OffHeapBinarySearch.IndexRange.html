001    package com.alexkasko.unsafe.offheap;
002    
003    import java.io.Serializable;
004    
005    /**
006     * Binary search implementation borrowed from {@code https://android.googlesource.com/platform/libcore/+/android-4.2.2_r1/luni/src/main/java/java/util/Arrays.java}
007     * and adapted to {@link OffHeapAddressable}
008     *
009     * @author alexkasko
010     * Date: 3/5/13
011     * @see OffHeapLongSorter
012     * @see OffHeapPayloadSorter
013     */
014    public class OffHeapBinarySearch {
015        /**
016         * Performs a binary search for {@code value} in the ascending sorted off-heap collection.
017         * Searching in an unsorted collection has an undefined result. It's also undefined which element
018         * is found if there are multiple occurrences of the same element.
019         *
020         * @param collection the sorted array to search.
021         * @param value the element to find.
022         * @return the non-negative index of the element, or a negative index which
023         *         is {@code -index - 1} where the element would be inserted.
024         */
025        public static long binarySearch(OffHeapAddressable collection, long value) {
026            return binarySearch(collection, 0, collection.size(), value);
027        }
028    
029        /**
030         * Performs a binary search for {@code value} in the ascending sorted off-heap collection,
031         * in the range specified by fromIndex (inclusive) and toIndex (exclusive).
032         * Searching in an unsorted collection has an undefined result. It's also undefined which element
033         * is found if there are multiple occurrences of the same element.
034         *
035         * @param collection      the sorted collection to search.
036         * @param startIndex the inclusive start index.
037         * @param endIndex   the exclusive end index.
038         * @param value      the element to find.
039         * @return the non-negative index of the element, or a negative index which
040         *         is {@code -index - 1} where the element would be inserted.
041         * @throws IllegalArgumentException {@code if (startIndex < 0 || startIndex > endIndex || endIndex > collection.size()}
042         */
043        public static long binarySearch(OffHeapAddressable collection, long startIndex, long endIndex, long value) {
044            if (startIndex < 0 || startIndex > endIndex || endIndex > collection.size()) {
045                throw new IllegalArgumentException("Illegal input, collection size: [" + collection.size() + "], " +
046                        "startIndex: [" + startIndex + "], endIndex: [" + endIndex + "]");
047            }
048            long lo = startIndex;
049            long hi = endIndex - 1;
050            while (lo <= hi) {
051                long mid = (lo + hi) >>> 1;
052                long midVal = collection.get(mid);
053    
054                if (midVal < value) {
055                    lo = mid + 1;
056                } else if (midVal > value) {
057                    hi = mid - 1;
058                } else {
059                    return mid;  // value found
060                }
061            }
062            return ~lo;  // value not present
063        }
064    
065        /**
066         * Performs a binary search for {@code value} in the ascending sorted off-heap collection.
067         * Returns range of indices having given value or empty range.
068         * Searching in an unsorted collection has an undefined result. It's also undefined which element
069         * is found if there are multiple occurrences of the same element.
070         *
071         * @param collection the sorted array to search.
072         * @param value the element to find.
073         * @return range of indices having given value or empty range
074         */
075        public static void binarySearchRange(OffHeapAddressable collection, long value, IndexRange out) {
076            binarySearchRange(collection, 0, collection.size(), value, out);
077        }
078    
079        /**
080         * Performs a binary search for {@code value} in the ascending sorted off-heap collection.
081         * Returns range of indices having given value or empty range.
082         * Searching in an unsorted collection has an undefined result. It's also undefined which element
083         * is found if there are multiple occurrences of the same element.
084         *
085         * @param collection the sorted array to search.
086         * @param startIndex the inclusive start index.
087         * @param endIndex   the exclusive end index.
088         * @param value the element to find.
089         * @return range of indices having given value or empty range
090         */
091        public static void binarySearchRange(OffHeapAddressable collection, long startIndex, long endIndex,
092                                                   long value, IndexRange out) {
093            long ind = binarySearch(collection, startIndex, endIndex, value);
094            if(ind < 0) {
095                out.setEmpty(ind);
096                return;
097            }
098            long from = ind;
099            while (from >= startIndex && value == collection.get(from)) from -= 1;
100            from += 1;
101            long to = ind;
102            while (to < endIndex && value == collection.get(to)) to += 1;
103            to -= 1;
104            out.set(from, to);
105        }
106    
107        /**
108         * {@link OffHeapAddressable} index range representation.
109         * {@link #isEmpty()} method should be checked before accessing indices.
110         * Empty range will contain negative equal indices which values are
111         * {@code -index - 1} where the element would be inserted.
112         * Range with size {@code 1} will contain equal indices.
113         */
114        public static class IndexRange implements Serializable {
115            private static final long serialVersionUID = 879348143026038919L;
116            private boolean empty;
117            private long fromIndex;
118            private long toIndex;
119    
120            public IndexRange() {
121                this.empty = false;
122                this.fromIndex = -1;
123                this.toIndex = -1;
124            }
125    
126            /**
127             * Empty range constructor
128             */
129            private void setEmpty(long value) {
130                this.empty = true;
131                this.fromIndex = value;
132                this.toIndex = value;
133            }
134    
135            /**
136             * Non-empty range constructor
137             *
138             * @param from start index
139             * @param to end index
140             */
141            private void set(long from, long to) {
142                this.empty = false;
143                this.fromIndex = from;
144                this.toIndex = to;
145            }
146    
147            /**
148             * Whether this range is empty
149             *
150             * @return whether this range is empty
151             */
152            public boolean isEmpty() {
153                return empty;
154            }
155    
156            /**
157             * Whether this range is not empty
158             *
159             * @return whether this range is not empty
160             */
161            public boolean isNotEmpty() {
162                return !empty;
163            }
164    
165            /**
166             * Start index or {@code -index - 1}
167             *      where the element would be inserted for empty range
168             *
169             * @return start index value or {@code -index - 1}
170             *      where the element would be inserted for empty range
171             */
172            public long getFromIndex() {
173                return fromIndex;
174            }
175    
176            /**
177             * End index or {@code -index - 1}
178             *      where the element would be inserted for empty range
179             *
180             * @return end index value, {@code -index - 1}
181             *      where the element would be inserted for empty range
182             */
183            public long getToIndex() {
184                return toIndex;
185            }
186    
187            /**
188             * {@inheritDoc}
189             */
190            @Override
191            public boolean equals(Object o) {
192                if (this == o) return true;
193                if (o == null || getClass() != o.getClass()) return false;
194                IndexRange that = (IndexRange) o;
195                if (empty != that.empty) return false;
196                if (fromIndex != that.fromIndex) return false;
197                if (toIndex != that.toIndex) return false;
198                return true;
199            }
200    
201            /**
202             * {@inheritDoc}
203             */
204            @Override
205            public int hashCode() {
206                int result = (empty ? 1 : 0);
207                result = 31 * result + (int) (fromIndex ^ (fromIndex >>> 32));
208                result = 31 * result + (int) (toIndex ^ (toIndex >>> 32));
209                return result;
210            }
211    
212            /**
213             * {@inheritDoc}
214             */
215            @Override
216            public String toString() {
217                final StringBuilder sb = new StringBuilder();
218                sb.append("IndexRange");
219                sb.append("{empty=").append(empty);
220                sb.append(", fromIndex=").append(fromIndex);
221                sb.append(", toIndex=").append(toIndex);
222                sb.append('}');
223                return sb.toString();
224            }
225        }
226    }
