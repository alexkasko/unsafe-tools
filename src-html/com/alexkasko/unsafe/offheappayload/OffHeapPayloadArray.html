017package com.alexkasko.unsafe.offheappayload;
018
019import com.alexkasko.unsafe.offheap.OffHeapDisposable;
020import com.alexkasko.unsafe.offheap.OffHeapMemory;
021
022/**
023 * Implementation of off-heap header-payload array. Memory block for each index
024 * contains {@code long} header anb {@code byte[]} payload.
025 *
026 * Default implementation uses {@code sun.misc.Unsafe}, with all operations guarded with {@code assert} keyword.
027 * With assertions enabled in runtime ({@code -ea} java switch) {@link AssertionError}
028 * will be thrown on illegal index access. Without assertions illegal index will crash JVM.
029 *
030 * Array won't be zeroed after creation (will contain garbage by default).
031 * Allocated memory may be freed manually using {@link #free()} (thread-safe
032 * and may be called multiple times) or it will be freed after {@link OffHeapPayloadArray} instance
033 * will be garbage collected.
034 *
035 * @author alexkasko
036 * Date: 3/3/13
037 */
038@Deprecated // use offheapstruct package
039public class OffHeapPayloadArray implements OffHeapPayloadAddressable, OffHeapDisposable {
040    private static final int HEADER_LENGTH = 8;
041
042    private final OffHeapMemory ohm;
043    private final int payloadLength;
044    private final int elementLength;
045
046    /**
047     * Constructor
048     *
049     * @param size array size
050     * @param payloadLength length of payload data block in bytes
051     */
052    public OffHeapPayloadArray(long size, int payloadLength) {
053        if(payloadLength <= 0) throw new IllegalArgumentException("Illegal payloadLength: [" + payloadLength + "]");
054        this.payloadLength = payloadLength;
055        this.elementLength = HEADER_LENGTH + payloadLength;
056        this.ohm = OffHeapMemory.allocateMemory(size * elementLength);
057    }
058
059    /**
060     * Whether unsafe implementation of {@link OffHeapMemory} is used
061     *
062     * @return whether unsafe implementation of {@link OffHeapMemory} is used
063     */
064    public boolean isUnsafe() {
065        return ohm.isUnsafe();
066    }
067
068    /**
069     * Returns length of the each payload block in bytes
070     *
071     * @return length of the each payload block in bytes
072     */
073    @Override
074    public int payloadLength() {
075        return payloadLength;
076    }
077
078    /**
079     * Gets the header at position {@code index}
080     *
081     * @param index collection index
082     * @return long value
083     */
084    @Override
085    public long get(long index) {
086        return ohm.getLong(index * elementLength);
087    }
088
089    /**
090     * Loads payload into provided buffer
091     *
092     * @param index collection index to load payload from
093     * @param buffer buffer to load payload into
094     */
095    @Override
096    public void getPayload(long index, byte[] buffer) {
097        ohm.get(index * elementLength + HEADER_LENGTH, buffer);
098    }
099
100    /**
101     * Loads payload into provided buffer
102     *
103     * @param index collection index to load payload from
104     * @param buffer buffer to load payload into
105     * @param bufferPos buffer offset
106     */
107    public void getPayload(long index, byte[] buffer, int bufferPos) {
108        ohm.get(index * elementLength + HEADER_LENGTH, buffer, bufferPos, payloadLength);
109    }
110
111    /**
112     * Sets header value and copies payload data on provided index
113     *
114     * @param index collection index to set header and payload
115     * @param header header value
116     * @param payload payload value
117     */
118    @Override
119    public void set(long index, long header, byte[] payload) {
120        long addr = index * elementLength;
121        ohm.putLong(addr, header);
122        ohm.put(addr + HEADER_LENGTH, payload);
123    }
124
125    /**
126     * Sets header value and copies payload data on provided index
127     *
128     * @param index collection index to set header and payload
129     * @param header header value
130     * @param payload payload value
131     * @param payloadPos payload offset
132     */
133    public void set(long index, long header, byte[] payload, int payloadPos) {
134        long addr = index * elementLength;
135        ohm.putLong(addr, header);
136        ohm.put(addr + HEADER_LENGTH, payload, payloadPos, payloadLength);
137    }
138
139    /**
140     * Copies payload data on provided index
141     *
142     * @param index collection index to set payload
143     * @param payload payload value
144     */
145    public void setPayload(long index, byte[] payload) {
146        long addr = index * elementLength;
147        ohm.put(addr + HEADER_LENGTH, payload);
148    }
149
150    /**
151     * Copies payload data on provided index
152     *
153     * @param index collection index to set payload
154     * @param payload payload value
155     * @param payloadPos payload offset
156     */
157    public void setPayload(long index, byte[] payload, int payloadPos) {
158        long addr = index * elementLength;
159        ohm.put(addr + HEADER_LENGTH, payload, payloadPos, payloadLength);
160    }
161
162    /**
163     * Returns number of elements in array
164     *
165     * @return number of elements in array
166     */
167    @Override
168    public long size() {
169        return ohm.length() / elementLength;
170    }
171
172    /**
173     * Frees allocated memory, may be called multiple times from any thread
174     */
175    @Override
176    public void free() {
177        ohm.free();
178    }
179
180    /**
181     * {@inheritDoc}
182     */
183    @Override
184    public String toString() {
185        final StringBuilder sb = new StringBuilder();
186        sb.append("OffHeapPayloadArray");
187        sb.append("{size=").append(size());
188        sb.append(", unsafe=").append(isUnsafe());
189        sb.append(", payloadLength=").append(payloadLength);
190        sb.append('}');
191        return sb.toString();
192    }
193}
