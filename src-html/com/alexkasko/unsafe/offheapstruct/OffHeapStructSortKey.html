017    package com.alexkasko.unsafe.offheapstruct;

019    import java.io.Serializable;
020    
021    /**
022     * Struct sort key class to use for sorting with multiple keys
023     *
024     * @author alexkasko
025     * Date: 7/8/13
026     */
027    public abstract class OffHeapStructSortKey implements Serializable {
028        private static final long serialVersionUID = 6811864689863999749L;
029    
030        private final int offset;
031    
032        /**
033         * Protected constructor for inheritors.
034         *
035         * @param offset key field offset in struct
036         */
037        protected OffHeapStructSortKey(int offset) {
038            if(offset < 0) throw new IllegalArgumentException("Specified offset: [" + offset + "] must not be negative");
039            this.offset = offset;
040        }
041    
042        /**
043         * Key type id for internal use
044         *
045         * @return key type id
046         */
047        abstract byte typeId();
048    
049        /**
050         * Factory method for struct sort key implementation
051         * that uses signed byte comparison on specified struct offset
052         *
053         * @param offset struct key offset
054         * @return sort key instance
055         */
056        public static OffHeapStructSortKey byteSortKey(int offset) {
057            return new ByteKey(offset);
058        }
059    
060        /**
061         * Factory method for struct sort key implementation
062         * that uses unsigned byte comparison on specified struct offset
063         *
064         * @param offset struct key offset
065         * @return sort key instance
066         */
067        public static OffHeapStructSortKey unsignedByteSortKey(int offset) {
068            return new UnsignedByteKey(offset);
069        }
070    
071        /**
072         * Factory method for struct sort key implementation
073         * that uses signed short comparison on specified struct offset
074         *
075         * @param offset struct key offset
076         * @return sort key instance
077         */
078        public static OffHeapStructSortKey shortSortKey(int offset) {
079            return new ShortKey(offset);
080        }
081    
082        /**
083         * Factory method for struct sort key implementation
084         * that uses unsigned short comparison on specified struct offset
085         *
086         * @param offset struct key offset
087         * @return sort key instance
088         */
089        public static OffHeapStructSortKey unsignedShortSortKey(int offset) {
090            return new UnsignedShortKey(offset);
091        }
092    
093        /**
094         * Factory method for struct sort key implementation
095         * that uses signed int comparison on specified struct offset
096         *
097         * @param offset struct key offset
098         * @return sort key instance
099         */
100        public static OffHeapStructSortKey intSortKey(int offset) {
101            return new IntKey(offset);
102        }
103    
104        /**
105         * Factory method for struct sort key implementation
106         * that uses unsigned int comparison on specified struct offset
107         *
108         * @param offset struct key offset
109         * @return sort key instance
110         */
111        public static OffHeapStructSortKey unsignedIntSortKey(int offset) {
112            return new UnsignedIntKey(offset);
113        }
114    
115        /**
116         * Factory method for struct sort key implementation
117         * that uses signed long comparison on specified struct offset
118         *
119         * @param offset struct key offset
120         * @return sort key instance
121         */
122        public static OffHeapStructSortKey longSortKey(int offset) {
123            return new LongKey(offset);
124        }
125    
126        /**
127         * Factory method for struct sort key implementation
128         * that uses unsigned long comparison on specified struct offset
129         *
130         * @param offset struct key offset
131         * @return sort key instance
132         */
133        public static OffHeapStructSortKey unsignedLongSortKey(int offset) {
134            return new UnsignedLongKey(offset);
135        }
136    
137        /**
138         * Offset accessor
139         *
140         * @return key offset
141         */
142        public int offset() {
143            return offset;
144        }
145    
146        /**
147         * {@inheritDoc}
148         */
149        @Override
150        public boolean equals(Object o) {
151            if (this == o) return true;
152            if (!(o instanceof OffHeapStructSortKey)) return false;
153            OffHeapStructSortKey that = (OffHeapStructSortKey) o;
154            if (offset != that.offset) return false;
155            return true;
156        }
157    
158        /**
159         * {@inheritDoc}
160         */
161        @Override
162        public int hashCode() {
163            return offset;
164        }
165    
166        /**
167         * {@inheritDoc}
168         */
169        @Override
170        public String toString() {
171            final StringBuilder sb = new StringBuilder();
172            sb.append(getClass().getSimpleName());
173            sb.append("{offset=").append(offset);
174            sb.append('}');
175            return sb.toString();
176        }
177    
178        /**
179         * Struct sort key implementation, uses signed byte comparison on specified struct offset
180         */
181        public static class ByteKey extends OffHeapStructSortKey {
182            private static final long serialVersionUID = -5201378839882085844L;
183            static final byte ID = 1;
184    
185            /**
186             * Constructor
187             *
188             * @param offset key field offset in struct
189             */
190            public ByteKey(int offset) {
191                super(offset);
192            }
193    
194            /**
195             * {@inheritDoc}
196             */
197            byte typeId() {
198                return ID;
199            }
200        }
