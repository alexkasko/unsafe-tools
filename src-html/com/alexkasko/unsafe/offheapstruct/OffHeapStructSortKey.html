017    package com.alexkasko.unsafe.offheapstruct;
018    
019    import java.io.Serializable;
020    import java.util.LinkedHashMap;
021    import java.util.Map;
022    
023    /**
024     * Helper container for struct sort key. Contains key offset and type:
025     * {@code long.class}, {@code int.class}, {@code short.class} or {@code byte.class}
026     *
027     * @author alexkasko
028     * Date: 7/8/13
029     */
030    public class OffHeapStructSortKey implements Serializable {
031        private static final long serialVersionUID = 6811864689863999749L;
032        private static final Map<Class<? extends Number>, Integer> KEY_TYPES;
033    
034        static {
035            KEY_TYPES = new LinkedHashMap<Class<? extends Number>, Integer>();
036            KEY_TYPES.put(long.class, 8);
037            KEY_TYPES.put(int.class, 4);
038            KEY_TYPES.put(short.class, 2);
039            KEY_TYPES.put(byte.class, 1);
040        }
041    
042        private final int offset;
043        private final int len;
044    
045        /**
046         * Constructor. Allowed key types are:
047         * {@code long.class}, {@code int.class}, {@code short.class} or {@code byte.class}
048         *
049         * @param offset key offset
050         * @param type key type
051         */
052        public OffHeapStructSortKey(int offset, Class<? extends Number> type) {
053            if(null == type) throw new IllegalArgumentException("Specified type is null");
054            if(!KEY_TYPES.containsKey(type)) throw new IllegalArgumentException(
055                    "Unsupported type, supported types are: [" + KEY_TYPES.keySet() + "]");
056            this.offset = offset;
057            this.len = KEY_TYPES.get(type);
058        }
059    
060        /**
061         * Offset accessor
062         *
063         * @return key offset
064         */
065        public int offset() {
066            return offset;
067        }
068    
069        /**
070         * Type length accessor
071         *
072         * @return key type length in bytes
073         */
074        public int length() {
075            return len;
076        }
077    
078        /**
079         * {@inheritDoc}
080         */
081        @Override
082        public boolean equals(Object o) {
083            if (this == o) return true;
084            if (o == null || getClass() != o.getClass()) return false;
085            OffHeapStructSortKey that = (OffHeapStructSortKey) o;
086            if (len != that.len) return false;
087            if (offset != that.offset) return false;
088            return true;
089        }
090    
091        /**
092         * {@inheritDoc}
093         */
094        @Override
095        public int hashCode() {
096            int result = offset;
097            result = 31 * result + len;
098            return result;
099        }
100    
101        /**
102         * {@inheritDoc}
103         */
104        @Override
105        public String toString() {
106            final StringBuilder sb = new StringBuilder();
107            sb.append("OffHeapStructSortCondition");
108            sb.append("{offset=").append(offset);
109            sb.append(", len=").append(len);
110            sb.append('}');
111            return sb.toString();
112        }
113    }
