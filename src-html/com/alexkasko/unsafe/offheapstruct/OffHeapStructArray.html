017    package com.alexkasko.unsafe.offheapstruct;
018    
019    import com.alexkasko.unsafe.offheap.OffHeapDisposable;
020    import com.alexkasko.unsafe.offheap.OffHeapMemory;
021    
022    import java.util.Iterator;
023    
024    /**
025     * <p>Implementation of off-heap array of structs (memory areas of equal sizes).
026     *
027     * <p>Default implementation uses {@code sun.misc.Unsafe}, with all operations guarded with {@code assert} keyword.
028     * With assertions enabled in runtime ({@code -ea} java switch) {@link AssertionError}
029     * will be thrown on illegal index access. Without assertions illegal index will crash JVM.
030     *
031     * <p>Array won't be zeroed after creation (will contain garbage by default).
032     * Allocated memory may be freed manually using {@link #free()} (thread-safe
033     * and may be called multiple times) or it will be freed after {@link com.alexkasko.unsafe.offheapstruct.OffHeapStructArray}
034     * instance will be garbage collected.
035     *
036     * @author alexkasko
037     * Date: 7/3/13
038     */
039    public class OffHeapStructArray implements OffHeapStructCollection, OffHeapDisposable, Iterable<byte[]> {
040    
041        private final OffHeapMemory ohm;
042        private final int structLength;
043    
044        /**
045          * Constructor
046          *
047          * @param size array size
048          * @param structLength length of struct in bytes, must be >= {@code 8}
049          */
050        public OffHeapStructArray(long size, int structLength) {
051            if(structLength < 8) throw new IllegalArgumentException(
052                    "structLength must be greater or equal to 8, but was: [" + structLength + "]");
053            this.structLength = structLength;
054            this.ohm = OffHeapMemory.allocateMemory(size * structLength);
055        }
056    
057        /**
058         * Returns length of the single struct in bytes
059         *
060         * @return length of the single struct in bytes
061         */
062        @Override
063        public int structLength() {
064            return structLength;
065        }
066    
067        /**
068         * Returns number of elements in this array
069         *
070         * @return number of elements in this array
071         */
072        @Override
073        public long size() {
074            return ohm.length() / structLength;
075        }
076    
077        /**
078         * Frees allocated memory, may be called multiple times from any thread
079         */
080        @Override
081        public void free() {
082            ohm.free();
083        }
084    
085        /**
086         * {@inheritDoc}
087         */
088        @Override
089        public Iterator<byte[]> iterator() {
090            return new OffHeapStructIterator(this);
091        }
092    
093        /**
094         * Whether unsafe implementation of {@link OffHeapMemory} is used
095         *
096         * @return whether unsafe implementation of {@link OffHeapMemory} is used
097         */
098        public boolean isUnsafe() {
099            return ohm.isUnsafe();
100        }
101    
102        /**
103         * Copies struct on specified index into specified buffer
104         *
105         * @param index array index
106         * @param buffer buffer to copy struct into
107         */
108        @Override
109        public void get(long index, byte[] buffer) {
110            ohm.get(index * structLength, buffer);
111        }
112    
113        /**
114         * Copies struct on specified index into specified buffer
115         *
116         * @param index array index
117         * @param buffer buffer to copy struct into
118         * @param bufferPos start position in specified buffer
119         */
120        public void get(long index, byte[] buffer, int bufferPos) {
121            ohm.get(index * structLength, buffer, bufferPos, structLength);
122        }
123    
124        /**
125         * Copies specified struct contents onto specified index
126         *
127         * @param index array index
128         * @param struct struct to copy into array
129         */
130        @Override
131        public void set(long index, byte[] struct) {
132            ohm.put(index * structLength, struct);
133        }
134    
135        /**
136         * Copies specified struct contents onto specified index
137         *
138         * @param index array index
139         * @param struct struct to copy into array
140         * @param structPos start position in specified struct
141         */
142        public void set(long index, byte[] struct, int structPos) {
143            ohm.put(index * structLength, struct, structPos, structLength);
144        }
145    
146        /**
147         * Gets byte from struct on specified index with specified offset
148         *
149         * @param index  array index
150         * @param offset struct offset
151         * @return byte
152         */
153        @Override
154        public byte getByte(long index, int offset) {
155            assert offset <= structLength - 1 : offset;
156            return ohm.getByte(index * structLength + offset);
157        }
158    
159        /**
160         * Puts byte into struct onto specified index with specified offset
161         *
162         * @param index  array index
163         * @param offset struct offset
164         * @param value  value
165         */
166        @Override
167        public void putByte(long index, int offset, byte value) {
168            assert offset <= structLength - 1 : offset;
169            ohm.putByte(index * structLength + offset, value);
170        }
171    
172        /**
173         * Gets one byte (stored as one signed byte) from struct on specified index
174         * with specified offset, converts it to unsigned and returns it as short
175         *
176         * @param index  array index
177         * @param offset struct offset
178         * @return unsigned byte as short
179         */
180        @Override
181        public short getUnsignedByte(long index, int offset) {
182            assert offset <= structLength - 1 : offset;
183            return ohm.getUnsignedByte(index * structLength + offset);
184        }
185    
186        /**
187         * Puts short with value from 0 to 255 inclusive into struct onto specified
188         * index with specified offset as one signed byte
189         *
190         * @param index  array index
191         * @param offset struct offset
192         * @param value  unsigned byte
193         */
194        @Override
195        public void putUnsignedByte(long index, int offset, short value) {
196            assert offset <= structLength - 1 : offset;
197            ohm.putUnsignedByte(index * structLength + offset, value);
198        }
199    
200        /**
201         * Gets two bytes as short from struct on specified index with specified offset
202         *
203         * @param index  array index
204         * @param offset struct offset
205         * @return short value
206         */
207        @Override
208        public short getShort(long index, int offset) {
209            assert offset <= structLength - 2 : offset;
210            return ohm.getShort(index * structLength + offset);
211        }
212    
213        /**
214         * Puts short into struct onto specified index with specified offset as two bytes
215         *
216         * @param index  array index
217         * @param offset struct offset
218         * @param value  short value
219         */
220        @Override
221        public void putShort(long index, int offset, short value) {
222            assert offset <= structLength - 2 : offset;
223            ohm.putShort(index * structLength + offset, value);
224        }
225    
226        /**
227         * Gets unsigned short (stored as two bytes) from struct on specified index
228         * with specified offset and returns it as int
229         *
230         * @param index  array index
231         * @param offset struct offset
232         * @return unsigned short as int
233         */
234        @Override
235        public int getUnsignedShort(long index, int offset) {
236            assert offset <= structLength - 2 : offset;
237            return ohm.getUnsignedShort(index * structLength + offset);
238        }
239    
240        /**
241         * Puts int with value from 0 to 65535 inclusive into struct onto specified
242         * index with specified offset as two bytes
243         *
244         * @param index  array index
245         * @param offset struct offset
246         * @param value  unsigned short as int
247         */
248        @Override
249        public void putUnsignedShort(long index, int offset, int value) {
250            assert offset <= structLength - 2 : offset;
251            ohm.putUnsignedShort(index * structLength + offset, value);
252        }
253    
254        /**
255         * Gets four bytes as int from struct on specified index with specified offset
256         *
257         * @param index  array index
258         * @param offset struct offset
259         * @return int value
260         */
261        @Override
262        public int getInt(long index, int offset) {
263            assert offset <= structLength - 4 : offset;
264            return ohm.getInt(index * structLength + offset);
265        }
266    
267        /**
268         * Puts int into struct onto specified index with specified offset as four bytes
269         *
270         * @param index  array index
271         * @param offset struct offset
272         * @param value  int value
273         */
274        @Override
275        public void putInt(long index, int offset, int value) {
276            assert offset <= structLength - 4 : offset;
277            ohm.putInt(index * structLength + offset, value);
278        }
279    
280        /**
281         * Gets unsigned int (stored as 4 bytes) and returns it as long
282         * from struct on specified index with specified offset
283         *
284         * @param index  array index
285         * @param offset struct offset
286         * @return unsigned int as long
287         */
288        @Override
289        public long getUnsignedInt(long index, int offset) {
290            assert offset <= structLength - 4 : offset;
291            return ohm.getUnsignedInt(index * structLength + offset);
292        }
293    
294        /**
295         * Puts long value from 0 to 4294967295 inclusive into struct onto specified index
296         * with specified offset as four bytes
297         *
298         * @param index  array index
299         * @param offset struct offset
300         * @param value  unsigned int as long
301         */
302        @Override
303        public void putUnsignedInt(long index, int offset, long value) {
304            assert offset <= structLength - 4 : offset;
305            ohm.putUnsignedInt(index * structLength + offset, value);
306        }
307    
308        /**
309         * Gets long from struct on specified index with specified offset
310         *
311         * @param index  array index
312         * @param offset struct offset
313         * @return long value
314         */
315        @Override
316        public long getLong(long index, int offset) {
317            assert offset <= structLength - 8 : offset;
318            return ohm.getLong(index * structLength + offset);
319        }
320    
321        /**
322         * Puts long into struct onto specified index with specified offset as eight bytes
323         *
324         * @param index  array index
325         * @param offset struct offset
326         * @param value  long value
327         */
328        @Override
329        public void putLong(long index, int offset, long value) {
330            assert offset <= structLength - 8 : offset;
331            ohm.putLong(index * structLength + offset, value);
332        }
333    
334        /**
335         * {@inheritDoc}
336         */
337        @Override
338        public String toString() {
339            final StringBuilder sb = new StringBuilder();
340            sb.append("OffHeapStructArray");
341            sb.append("{ohm=").append(ohm);
342            sb.append(", structLength=").append(structLength);
343            sb.append('}');
344            return sb.toString();
345        }
346    }
