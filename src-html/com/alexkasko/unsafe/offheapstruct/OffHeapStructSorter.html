019    /**
020     * <p>alexkasko: borrowed from {@code https://android.googlesource.com/platform/libcore/+/android-4.2.2_r1/luni/src/main/java/java/util/DualPivotQuicksort.java}
021     * and adapted to {@link OffHeapStructCollection}.
022     *
023     * <p>This class implements the Dual-Pivot Quicksort algorithm by
024     * Vladimir Yaroslavskiy, Jon Bentley, and Joshua Bloch. The algorithm
025     * offers O(n log(n)) performance on many data sets that cause other
026     * quicksorts to degrade to quadratic performance, and is typically
027     * faster than traditional (one-pivot) Quicksort implementations.
028     *
029     * @author Vladimir Yaroslavskiy
030     * @author Jon Bentley
031     * @author Josh Bloch
032     *
033     * @version 2009.11.29 m765.827.12i
034     */

038         * If the length of an collection to be sorted is less than this
039         * constant, insertion sort is used in preference to Quicksort.
040         */
041        private static final int INSERTION_SORT_THRESHOLD = 32;

051         * @param keyOffset int key field offset within stuct bounds

053        public static void sortByIntKey(OffHeapStructCollection a, int keyOffset) {
054            sortByIntKey(a, 0, a.size(), keyOffset);

057        /**
058         * Sorts the specified range of the off-heap struct collection into ascending order using int struct key. The range
059         * to be sorted extends from the index {@code fromIndex}, inclusive, to
060         * the index {@code toIndex}, exclusive. If {@code fromIndex == toIndex},
061         * the range to be sorted is empty (and the call is a no-op).
062         *
063         * @param a the off-heap struct collection to be sorted
064         * @param fromIndex the index of the first element, inclusive, to be sorted
065         * @param toIndex the index of the last element, exclusive, to be sorted
066         * @param keyOffset int sort key field offset within stuct bounds
067         * @throws IllegalArgumentException {@code if (fromIndex < 0 || fromIndex > toIndex || toIndex > a.size())}
068         */
069        public static void sortByIntKey(OffHeapStructCollection a, long fromIndex, long toIndex, int keyOffset) {
070            if (fromIndex < 0 || fromIndex > toIndex || toIndex > a.size()) {
071                throw new IllegalArgumentException("Illegal input, collection size: [" + a.size() + "], " +
072                        "fromIndex: [" + fromIndex + "], toIndex: [" + toIndex + "]");
073            }
074            int len = a.structLength();
075            doSortByIntKey(a, fromIndex, toIndex - 1, keyOffset, new byte[len], new byte[len], new byte[len], new byte[len], new byte[len],
076                    new byte[len], new byte[len]);
077        }

080        /**
081         * Sorts the specified range of the off-heap struct collection into ascending order using int struct key.
082         * This method differs from the public {@code sort} method in that the
083         * {@code right} index is inclusive, and it does no range checking on
084         * {@code left} or {@code right}.
085         *
086         * @param a the off-heap header-payload collection to be sorted
087         * @param left the index of the first element, inclusive, to be sorted
088         * @param right the index of the last element, inclusive, to be sorted* @param keyOffset
089         * @param keyOffset int sort key field offset within stuct bounds
090         * @param pi temporary buffer for structs
091         * @param pj temporary buffer for structs
092         * @param pe1 temporary buffer for structs
093         * @param pe2 temporary buffer for structs
094         * @param pe3 temporary buffer for structs
095         * @param pe4 temporary buffer for structs
096         * @param pe5 temporary buffer for structs
097         */
098        private static void doSortByIntKey(OffHeapStructCollection a, long left, long right, int keyOffset, byte[] pi, byte[] pj,
099                                   byte[] pe1, byte[] pe2, byte[] pe3, byte[] pe4, byte[] pe5) {
100            // Use insertion sort on tiny arrays
101            if (right - left + 1 < INSERTION_SORT_THRESHOLD) {
102                for (long i = left + 1; i <= right; i++) {
103                    long ai = a.getInt(i, keyOffset);
104                    a.get(i, pi);
105                    long j;
106                    for (j = i - 1; j >= left && ai < a.getInt(j, keyOffset); j--) {
107                        a.get(j, pj);
108                        a.set(j + 1, pj);
109                    }
110                    a.set(j + 1, pi);
111                }
112            } else { // Use Dual-Pivot Quicksort on large arrays
113                dualPivotQuicksortByIntKey(a, left, right, keyOffset, pi, pj, pe1, pe2, pe3, pe4, pe5);
114            }
115        }

117        /**
118         * Sorts the specified range of the off-heap struct collection into ascending order by the
119         * Dual-Pivot Quicksort algorithm using int struct key.
120         *
121         * @param a the off-heap header-payload collection to be sorted
122         * @param left the index of the first element, inclusive, to be sorted
123         * @param right the index of the last element, inclusive, to be sorted
124         * @param keyOffset int sort key field offset within stuct bounds
125         * @param pi temporary buffer for structs
126         * @param pj temporary buffer for structs
127         * @param pe1 temporary buffer for structs
128         * @param pe2 temporary buffer for structs
129         * @param pe3 temporary buffer for structs
130         * @param pe4 temporary buffer for structs
131         * @param pe5 temporary buffer for structs
132         */
133        private static void dualPivotQuicksortByIntKey(OffHeapStructCollection a, long left, long right, int keyOffset, byte[] pi, byte[] pj,
134                                               byte[] pe1, byte[] pe2, byte[] pe3, byte[] pe4, byte[] pe5) {
135            // Compute indices of five evenly spaced elements
136            long sixth = (right - left + 1) / 6;
137            long e1 = left  + sixth;
138            long e5 = right - sixth;
139            long e3 = (left + right) >>> 1; // The midpoint
140            long e4 = e3 + sixth;
141            long e2 = e3 - sixth;
142    
143            // Sort these elements using a 5-element sorting network
144            long ae1 = a.getInt(e1, keyOffset), ae2 = a.getInt(e2, keyOffset), ae3 = a.getInt(e3, keyOffset),
145                    ae4 = a.getInt(e4, keyOffset), ae5 = a.getInt(e5, keyOffset);
146            a.get(e1, pe1); a.get(e2, pe2); a.get(e3, pe3); a.get(e4, pe4); a.get(e5, pe5);
147    
148            if (ae1 > ae2) { long t = ae1; byte[] pt = pe1; ae1 = ae2; pe1 = pe2; ae2 = t; pe2 = pt; }
149            if (ae4 > ae5) { long t = ae4; byte[] pt = pe4; ae4 = ae5; pe4 = pe5; ae5 = t; pe5 = pt; }
150            if (ae1 > ae3) { long t = ae1; byte[] pt = pe1; ae1 = ae3; pe1 = pe3; ae3 = t; pe3 = pt; }
151            if (ae2 > ae3) { long t = ae2; byte[] pt = pe2; ae2 = ae3; pe2 = pe3; ae3 = t; pe3 = pt; }
152            if (ae1 > ae4) { long t = ae1; byte[] pt = pe1; ae1 = ae4; pe1 = pe4; ae4 = t; pe4 = pt; }
153            if (ae3 > ae4) { long t = ae3; byte[] pt = pe3; ae3 = ae4; pe3 = pe4; ae4 = t; pe4 = pt; }
154            if (ae2 > ae5) { long t = ae2; byte[] pt = pe2; ae2 = ae5; pe2 = pe5; ae5 = t; pe5 = pt; }
155            if (ae2 > ae3) { long t = ae2; byte[] pt = pe2; ae2 = ae3; pe2 = pe3; ae3 = t; pe3 = pt; }
156            if (ae4 > ae5) { long t = ae4; byte[] pt = pe4; ae4 = ae5; pe4 = pe5; ae5 = t; pe5 = pt; }
157    
158            a.set(e1, pe1); a.set(e3, pe3); a.set(e5, pe5);
159    
160            /*
161             * Use the second and fourth of the five sorted elements as pivots.
162             * These values are inexpensive approximations of the first and
163             * second terciles of the array. Note that pivot1 <= pivot2.
164             *
165             * The pivots are stored in local variables, and the first and
166             * the last of the elements to be sorted are moved to the locations
167             * formerly occupied by the pivots. When partitioning is complete,
168             * the pivots are swapped back into their final positions, and
169             * excluded from subsequent sorting.
170             */
171            a.get(left, pe1);
172            long pivot1 = ae2; a.set(e2, pe1);
173            a.get(right, pe1);
174            long pivot2 = ae4; a.set(e4, pe1);
175    
176            // Pointers
177            long less  = left  + 1; // The index of first element of center part
178            long great = right - 1; // The index before first element of right part
179    
180            boolean pivotsDiffer = (pivot1 != pivot2);
181    
182            if (pivotsDiffer) {
183                /*
184                 * Partitioning:
185                 *
186                 *   left part         center part                    right part
187                 * +------------------------------------------------------------+
188                 * | < pivot1  |  pivot1 <= && <= pivot2  |    ?    |  > pivot2 |
189                 * +------------------------------------------------------------+
190                 *              ^                          ^       ^
191                 *              |                          |       |
192                 *             less                        k     great
193                 *
194                 * Invariants:
195                 *
196                 *              all in (left, less)   < pivot1
197                 *    pivot1 <= all in [less, k)     <= pivot2
198                 *              all in (great, right) > pivot2
199                 *
200                 * Pointer k is the first index of ?-part
201                 */
202                outer:
203                for (long k = less; k <= great; k++) {
204                    long ak = a.getInt(k, keyOffset);
205                    a.get(k, pe1);
206                    if (ak < pivot1) { // Move a[k] to left part
207                        if (k != less) {
208                            a.get(less, pe3);
209                            a.set(k, pe3);
210                            a.set(less, pe1);
211                        }
212                        less++;
213                    } else if (ak > pivot2) { // Move a[k] to right part
214                        while (a.getInt(great, keyOffset) > pivot2) {
215                            if (great-- == k) {
216                                break outer;
217                            }
218                        }
219                        if (a.getInt(great, keyOffset) < pivot1) {
220                            a.get(less, pe3);
221                            a.set(k, pe3);
222                            a.get(great, pe3);
223                            a.set(less++, pe3);
224                            a.set(great--, pe1);
225                        } else { // pivot1 <= a[great] <= pivot2
226                            a.get(great, pe3);
227                            a.set(k, pe3);
228                            a.set(great--, pe1);
229                        }
230                    }
231                }
232            } else { // Pivots are equal
233                /*
234                 * Partition degenerates to the traditional 3-way,
235                 * or "Dutch National Flag", partition:
236                 *
237                 *   left part   center part            right part
238                 * +----------------------------------------------+
239                 * |  < pivot  |  == pivot  |    ?    |  > pivot  |
240                 * +----------------------------------------------+
241                 *              ^            ^       ^
242                 *              |            |       |
243                 *             less          k     great
244                 *
245                 * Invariants:
246                 *
247                 *   all in (left, less)   < pivot
248                 *   all in [less, k)     == pivot
249                 *   all in (great, right) > pivot
250                 *
251                 * Pointer k is the first index of ?-part
252                 */
253                for (long k = less; k <= great; k++) {
254                    long ak = a.getInt(k, keyOffset);
255                    a.get(k, pe1);
256                    if (ak == pivot1) {
257                        continue;
258                    }
259                    if (ak < pivot1) { // Move a[k] to left part
260                        if (k != less) {
261                            a.get(less, pe3);
262                            a.set(k, pe3);
263                            a.set(less, pe1);
264                        }
265                        less++;
266                    } else { // (a[k] > pivot1) -  Move a[k] to right part
267                        /*
268                         * We know that pivot1 == a[e3] == pivot2. Thus, we know
269                         * that great will still be >= k when the following loop
270                         * terminates, even though we don't test for it explicitly.
271                         * In other words, a[e3] acts as a sentinel for great.
272                         */
273                        while (a.getInt(great, keyOffset) > pivot1) {
274                            great--;
275                        }
276                        if (a.getInt(great, keyOffset) < pivot1) {
277                            a.get(less, pe3);
278                            a.set(k, pe3);
279                            a.get(great, pe3);
280                            a.set(less++, pe3);
281                            a.set(great--, pe1);
282                        } else { // a[great] == pivot1
283                            a.get(great, pe3);
284                            a.set(k, pe3);
285                            a.set(great--, pe1);
286                        }
287                    }
288                }
289            }
290    
291            // Swap pivots into their final positions
292            a.get(less - 1, pe1);
293            a.set(left, pe1); a.set(less - 1, pe2);
294            a.get(great + 1, pe1);
295            a.set(right, pe1); a.set(great + 1, pe4);
296    
297            // Sort left and right parts recursively, excluding known pivot values
298            doSortByIntKey(a, left,   less - 2, keyOffset, pi, pj, pe1, pe2, pe3, pe4, pe5);
299            doSortByIntKey(a, great + 2, right, keyOffset, pi, pj, pe1, pe2, pe3, pe4, pe5);
300    
301            /*
302             * If pivot1 == pivot2, all elements from center
303             * part are equal and, therefore, already sorted
304             */
305            if (!pivotsDiffer) {
306                return;
307            }
308    
309            /*
310             * If center part is too large (comprises > 2/3 of the array),
311             * swap internal pivot values to ends
312             */
313            if (less < e1 && great > e5) {
314                while (a.getInt(less, keyOffset) == pivot1) {
315                    less++;
316                }
317                while (a.getInt(great, keyOffset) == pivot2) {
318                    great--;
319                }
320    
321                /*
322                 * Partitioning:
323                 *
324                 *   left part       center part                   right part
325                 * +----------------------------------------------------------+
326                 * | == pivot1 |  pivot1 < && < pivot2  |    ?    | == pivot2 |
327                 * +----------------------------------------------------------+
328                 *              ^                        ^       ^
329                 *              |                        |       |
330                 *             less                      k     great
331                 *
332                 * Invariants:
333                 *
334                 *              all in (*, less)  == pivot1
335                 *     pivot1 < all in [less, k)   < pivot2
336                 *              all in (great, *) == pivot2
337                 *
338                 * Pointer k is the first index of ?-part
339                 */
340                outer:
341                for (long k = less; k <= great; k++) {
342                    long ak = a.getInt(k, keyOffset);
343                    a.get(k, pe1);
344                    if (ak == pivot2) { // Move a[k] to right part
345                        while (a.getInt(great, keyOffset) == pivot2) {
346                            if (great-- == k) {
347                                break outer;
348                            }
349                        }
350                        if (a.getInt(great, keyOffset) == pivot1) {
351                            a.get(less, pe3);
352                            a.set(k, pe3);
353                            a.get(great, pe3);
354                            a.set(less++, pe3);
355                        } else { // pivot1 < a[great] < pivot2
356                            a.get(great, pe3);
357                            a.set(k, pe3);
358                        }
359                        a.set(great--, pe1);
360                    } else if (ak == pivot1) { // Move a[k] to left part
361                        a.get(less, pe3);
362                        a.set(k, pe3);
363                        a.set(less++, pe1);
364                    }
365                }
366            }
367    
368            // Sort center part recursively, excluding known pivot values
369            doSortByIntKey(a, less, great, keyOffset, pi, pj, pe1, pe2, pe3, pe4, pe5);
370        }

410         * Sorts the specified range of the off-heap struct collection into ascending order using long struct key.
411         * This method differs from the public {@code sort} method in that the
412         * {@code right} index is inclusive, and it does no range checking on
413         * {@code left} or {@code right}.
414         *
415         * @param a the off-heap header-payload collection to be sorted
416         * @param left the index of the first element, inclusive, to be sorted
417         * @param right the index of the last element, inclusive, to be sorted* @param keyOffset
418         * @param keyOffset long sort key field offset within stuct bounds
419         * @param pi temporary buffer for structs
420         * @param pj temporary buffer for structs
421         * @param pe1 temporary buffer for structs
422         * @param pe2 temporary buffer for structs
423         * @param pe3 temporary buffer for structs
424         * @param pe4 temporary buffer for structs
425         * @param pe5 temporary buffer for structs
426         */
427        private static void doSortByLongKey(OffHeapStructCollection a, long left, long right, int keyOffset, byte[] pi, byte[] pj,
428                                   byte[] pe1, byte[] pe2, byte[] pe3, byte[] pe4, byte[] pe5) {
429            // Use insertion sort on tiny arrays
430            if (right - left + 1 < INSERTION_SORT_THRESHOLD) {

437                        a.set(j + 1, pj);
438                    }
439                    a.set(j + 1, pi);
440                }
441            } else { // Use Dual-Pivot Quicksort on large arrays
442                dualPivotQuicksortByLongKey(a, left, right, keyOffset, pi, pj, pe1, pe2, pe3, pe4, pe5);
443            }

446        /**
447         * Sorts the specified range of the off-heap struct collection into ascending order by the
448         * Dual-Pivot Quicksort algorithm using long struct key.
449         *
450         * @param a the off-heap header-payload collection to be sorted
451         * @param left the index of the first element, inclusive, to be sorted
452         * @param right the index of the last element, inclusive, to be sorted
453         * @param keyOffset long sort key field offset within stuct bounds
454         * @param pi temporary buffer for structs
455         * @param pj temporary buffer for structs
456         * @param pe1 temporary buffer for structs
457         * @param pe2 temporary buffer for structs
458         * @param pe3 temporary buffer for structs
459         * @param pe4 temporary buffer for structs
460         * @param pe5 temporary buffer for structs
461         */
462        private static void dualPivotQuicksortByLongKey(OffHeapStructCollection a, long left, long right, int keyOffset, byte[] pi, byte[] pj,
463                                               byte[] pe1, byte[] pe2, byte[] pe3, byte[] pe4, byte[] pe5) {
464            // Compute indices of five evenly spaced elements
465            long sixth = (right - left + 1) / 6;
466            long e1 = left  + sixth;
467            long e5 = right - sixth;
468            long e3 = (left + right) >>> 1; // The midpoint
469            long e4 = e3 + sixth;
470            long e2 = e3 - sixth;
471    
472            // Sort these elements using a 5-element sorting network
473            long ae1 = a.getLong(e1, keyOffset), ae2 = a.getLong(e2, keyOffset), ae3 = a.getLong(e3, keyOffset),
474                    ae4 = a.getLong(e4, keyOffset), ae5 = a.getLong(e5, keyOffset);
475            a.get(e1, pe1); a.get(e2, pe2); a.get(e3, pe3); a.get(e4, pe4); a.get(e5, pe5);
476    
477            if (ae1 > ae2) { long t = ae1; byte[] pt = pe1; ae1 = ae2; pe1 = pe2; ae2 = t; pe2 = pt; }
478            if (ae4 > ae5) { long t = ae4; byte[] pt = pe4; ae4 = ae5; pe4 = pe5; ae5 = t; pe5 = pt; }
479            if (ae1 > ae3) { long t = ae1; byte[] pt = pe1; ae1 = ae3; pe1 = pe3; ae3 = t; pe3 = pt; }
480            if (ae2 > ae3) { long t = ae2; byte[] pt = pe2; ae2 = ae3; pe2 = pe3; ae3 = t; pe3 = pt; }
481            if (ae1 > ae4) { long t = ae1; byte[] pt = pe1; ae1 = ae4; pe1 = pe4; ae4 = t; pe4 = pt; }
482            if (ae3 > ae4) { long t = ae3; byte[] pt = pe3; ae3 = ae4; pe3 = pe4; ae4 = t; pe4 = pt; }
483            if (ae2 > ae5) { long t = ae2; byte[] pt = pe2; ae2 = ae5; pe2 = pe5; ae5 = t; pe5 = pt; }
484            if (ae2 > ae3) { long t = ae2; byte[] pt = pe2; ae2 = ae3; pe2 = pe3; ae3 = t; pe3 = pt; }
485            if (ae4 > ae5) { long t = ae4; byte[] pt = pe4; ae4 = ae5; pe4 = pe5; ae5 = t; pe5 = pt; }
486    
487            a.set(e1, pe1); a.set(e3, pe3); a.set(e5, pe5);
488    
489            /*
490             * Use the second and fourth of the five sorted elements as pivots.
491             * These values are inexpensive approximations of the first and
492             * second terciles of the array. Note that pivot1 <= pivot2.
493             *
494             * The pivots are stored in local variables, and the first and
495             * the last of the elements to be sorted are moved to the locations
496             * formerly occupied by the pivots. When partitioning is complete,
497             * the pivots are swapped back into their final positions, and
498             * excluded from subsequent sorting.
499             */
500            a.get(left, pe1);
501            long pivot1 = ae2; a.set(e2, pe1);
502            a.get(right, pe1);
503            long pivot2 = ae4; a.set(e4, pe1);
504    
505            // Pointers
506            long less  = left  + 1; // The index of first element of center part
507            long great = right - 1; // The index before first element of right part
508    
509            boolean pivotsDiffer = (pivot1 != pivot2);
510    
511            if (pivotsDiffer) {
512                /*
513                 * Partitioning:
514                 *
515                 *   left part         center part                    right part
516                 * +------------------------------------------------------------+
517                 * | < pivot1  |  pivot1 <= && <= pivot2  |    ?    |  > pivot2 |
518                 * +------------------------------------------------------------+
519                 *              ^                          ^       ^
520                 *              |                          |       |
521                 *             less                        k     great
522                 *
523                 * Invariants:
524                 *
525                 *              all in (left, less)   < pivot1
526                 *    pivot1 <= all in [less, k)     <= pivot2
527                 *              all in (great, right) > pivot2
528                 *
529                 * Pointer k is the first index of ?-part
530                 */
531                outer:
532                for (long k = less; k <= great; k++) {
533                    long ak = a.getLong(k, keyOffset);
534                    a.get(k, pe1);
535                    if (ak < pivot1) { // Move a[k] to left part
536                        if (k != less) {
537                            a.get(less, pe3);
538                            a.set(k, pe3);
539                            a.set(less, pe1);
540                        }
541                        less++;
542                    } else if (ak > pivot2) { // Move a[k] to right part
543                        while (a.getLong(great, keyOffset) > pivot2) {
544                            if (great-- == k) {
545                                break outer;
546                            }
547                        }
548                        if (a.getLong(great, keyOffset) < pivot1) {
549                            a.get(less, pe3);
550                            a.set(k, pe3);
551                            a.get(great, pe3);
552                            a.set(less++, pe3);
553                            a.set(great--, pe1);
554                        } else { // pivot1 <= a[great] <= pivot2
555                            a.get(great, pe3);
556                            a.set(k, pe3);
557                            a.set(great--, pe1);
558                        }
559                    }
560                }
561            } else { // Pivots are equal
562                /*
563                 * Partition degenerates to the traditional 3-way,
564                 * or "Dutch National Flag", partition:
565                 *
566                 *   left part   center part            right part
567                 * +----------------------------------------------+
568                 * |  < pivot  |  == pivot  |    ?    |  > pivot  |
569                 * +----------------------------------------------+
570                 *              ^            ^       ^
571                 *              |            |       |
572                 *             less          k     great
573                 *
574                 * Invariants:
575                 *
576                 *   all in (left, less)   < pivot
577                 *   all in [less, k)     == pivot
578                 *   all in (great, right) > pivot
579                 *
580                 * Pointer k is the first index of ?-part
581                 */
582                for (long k = less; k <= great; k++) {
583                    long ak = a.getLong(k, keyOffset);
584                    a.get(k, pe1);
585                    if (ak == pivot1) {
586                        continue;
587                    }
588                    if (ak < pivot1) { // Move a[k] to left part
589                        if (k != less) {
590                            a.get(less, pe3);
591                            a.set(k, pe3);
592                            a.set(less, pe1);
593                        }
594                        less++;
595                    } else { // (a[k] > pivot1) -  Move a[k] to right part
596                        /*
597                         * We know that pivot1 == a[e3] == pivot2. Thus, we know
598                         * that great will still be >= k when the following loop
599                         * terminates, even though we don't test for it explicitly.
600                         * In other words, a[e3] acts as a sentinel for great.
601                         */
602                        while (a.getLong(great, keyOffset) > pivot1) {
603                            great--;
604                        }
605                        if (a.getLong(great, keyOffset) < pivot1) {
606                            a.get(less, pe3);
607                            a.set(k, pe3);
608                            a.get(great, pe3);
609                            a.set(less++, pe3);
610                            a.set(great--, pe1);
611                        } else { // a[great] == pivot1
612                            a.get(great, pe3);
613                            a.set(k, pe3);
614                            a.set(great--, pe1);
615                        }
616                    }
617                }
618            }
619    
620            // Swap pivots into their final positions
621            a.get(less - 1, pe1);
622            a.set(left, pe1); a.set(less - 1, pe2);
623            a.get(great + 1, pe1);
624            a.set(right, pe1); a.set(great + 1, pe4);
625    
626            // Sort left and right parts recursively, excluding known pivot values
627            doSortByLongKey(a, left,   less - 2, keyOffset, pi, pj, pe1, pe2, pe3, pe4, pe5);
628            doSortByLongKey(a, great + 2, right, keyOffset, pi, pj, pe1, pe2, pe3, pe4, pe5);
629    
630            /*
631             * If pivot1 == pivot2, all elements from center
632             * part are equal and, therefore, already sorted
633             */
634            if (!pivotsDiffer) {
635                return;
636            }
637    
638            /*
639             * If center part is too large (comprises > 2/3 of the array),
640             * swap internal pivot values to ends
641             */
642            if (less < e1 && great > e5) {
643                while (a.getLong(less, keyOffset) == pivot1) {
644                    less++;
645                }
646                while (a.getLong(great, keyOffset) == pivot2) {
647                    great--;
648                }
649    
650                /*
651                 * Partitioning:
652                 *
653                 *   left part       center part                   right part
654                 * +----------------------------------------------------------+
655                 * | == pivot1 |  pivot1 < && < pivot2  |    ?    | == pivot2 |
656                 * +----------------------------------------------------------+
657                 *              ^                        ^       ^
658                 *              |                        |       |
659                 *             less                      k     great
660                 *
661                 * Invariants:
662                 *
663                 *              all in (*, less)  == pivot1
664                 *     pivot1 < all in [less, k)   < pivot2
665                 *              all in (great, *) == pivot2
666                 *
667                 * Pointer k is the first index of ?-part
668                 */
669                outer:
670                for (long k = less; k <= great; k++) {
671                    long ak = a.getLong(k, keyOffset);
672                    a.get(k, pe1);
673                    if (ak == pivot2) { // Move a[k] to right part
674                        while (a.getLong(great, keyOffset) == pivot2) {
675                            if (great-- == k) {
676                                break outer;
677                            }
678                        }
679                        if (a.getLong(great, keyOffset) == pivot1) {
680                            a.get(less, pe3);
681                            a.set(k, pe3);
682                            a.get(great, pe3);
683                            a.set(less++, pe3);
684                        } else { // pivot1 < a[great] < pivot2
685                            a.get(great, pe3);
686                            a.set(k, pe3);
687                        }
688                        a.set(great--, pe1);
689                    } else if (ak == pivot1) { // Move a[k] to left part
690                        a.get(less, pe3);
691                        a.set(k, pe3);
692                        a.set(less++, pe1);
693                    }
694                }
695            }
696    
697            // Sort center part recursively, excluding known pivot values
698            doSortByLongKey(a, less, great, keyOffset, pi, pj, pe1, pe2, pe3, pe4, pe5);
699        }
700    }
